     class PrideSoundSystem {
       constructor() {
         this.audioContext = null;
         this.sounds = {};
         this.isEnabled = false;
         this.init();
       }

       async init() {
         try {
           this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
           this.isEnabled = true;
           this.createSounds();
         } catch (error) {
           console.log('Audio not supported:', error);
         }
       }

       createSounds() {
         // Pride toggle celebration - the only sound we need!
         this.sounds.prideToggle = this.createTone([523.25, 659.25, 783.99, 1046.50, 1318.51], 1.5);
       }

       createTone(frequencies, duration) {
         return () => {
           if (!this.isEnabled || !this.audioContext) return;
           
           const now = this.audioContext.currentTime;
           frequencies.forEach((freq, i) => {
             const oscillator = this.audioContext.createOscillator();
             const gainNode = this.audioContext.createGain();
             
             oscillator.connect(gainNode);
             gainNode.connect(this.audioContext.destination);
             
             oscillator.frequency.setValueAtTime(freq, now + i * 0.1);
             oscillator.type = 'sine';
             
             gainNode.gain.setValueAtTime(0, now + i * 0.1);
             gainNode.gain.linearRampToValueAtTime(0.1, now + i * 0.1 + 0.01);
             gainNode.gain.exponentialRampToValueAtTime(0.001, now + i * 0.1 + duration / frequencies.length);
             
             oscillator.start(now + i * 0.1);
             oscillator.stop(now + i * 0.1 + duration / frequencies.length);
           });
         };
       }

       play(soundName) {
         if (this.sounds[soundName]) {
           this.sounds[soundName]();
           this.showAudioIndicator();
         }
       }

       showAudioIndicator() {
         const indicator = document.getElementById('audioIndicator');
         indicator.classList.add('show');
         setTimeout(() => indicator.classList.remove('show'), 1500);
       }
     }

     const BINGO_STORAGE_KEY = 'asslBingoProgress';
     const BINGO_VICTORY_THRESHOLD = 5;
     let tappedSponsors = new Set();
     let bingoComplete = false;

     function initSponsorBingo() {
        const sponsors = document.querySelectorAll('.sponsor-bingo');
        loadBingoProgress();

        sponsors.forEach(sponsor => {
            sponsor.addEventListener('click', handleSponsorTap);
        });

        const resetButton = document.getElementById('bingo-reset-btn');

            sessionStorage.removeItem(BINGO_STORAGE_KEY);

     function handleSponsorTap(event) {
        if (bingoComplete) return;

        const logo = event.currentTarget;
        const sponsorId = logo.id;

        if (!tappedSponsors.has(sponsorId)) {
            logo.classList.add('tapped');
            tappedSponsors.add(sponsorId);
            saveBingoProgress();
            checkBingoProgress();
        }
     }

     function checkBingoProgress() {
        if (tappedSponsors.size >= BINGO_VICTORY_THRESHOLD) {
            bingoComplete = true;
            createMassiveSparkleStorm();
            
            setTimeout(() => {
                alert("üéâ You completed Sponsor Bingo! Mahalo for supporting our league.");
            }, 1000);
        }
     }

     function loadBingoProgress() {
        const savedProgress = sessionStorage.getItem(BINGO_STORAGE_KEY);
        if (savedProgress) {
            const tappedIds = JSON.parse(savedProgress);
            tappedSponsors = new Set(tappedIds);
            
            tappedSponsors.forEach(sponsorId => {
                const logo = document.getElementById(sponsorId);
                if (logo) {
                    logo.classList.add('tapped');
                }
            });

            if (tappedSponsors.size >= BINGO_VICTORY_THRESHOLD) {
                bingoComplete = true;
            }
        }
     }

     function saveBingoProgress() {
        sessionStorage.setItem(BINGO_STORAGE_KEY, JSON.stringify([...tappedSponsors]));
     }
